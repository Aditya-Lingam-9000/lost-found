/* Lost & Found Campus System - Roll 1-3 Frontend Logic (no backend) */
(function () {
  const USERS_KEY = "lf_users";
  const SESSION_KEY = "lf_session";
  const LOST_KEY = "lf_lost_items";
  const FOUND_KEY = "lf_found_items";
  const MATCHES_KEY = "lf_matches";
  const CHATS_KEY = "lf_chats";
  const MESSAGES_KEY = "lf_messages";

  const pad4 = (n) => String(n).padStart(4, "0");

  const nowIso = () => new Date().toISOString();

  function safeJsonParse(raw, fallback) {
    try {
      return JSON.parse(raw);
    } catch {
      return fallback;
    }
  }

  function loadArray(key) {
    const raw = localStorage.getItem(key);
    if (!raw) return [];
    const parsed = safeJsonParse(raw, []);
    return Array.isArray(parsed) ? parsed : [];
  }

  function saveArray(key, arr) {
    localStorage.setItem(key, JSON.stringify(arr));
  }

  function getSession() {
    return safeJsonParse(localStorage.getItem(SESSION_KEY), null);
  }

  function setSession(session) {
    localStorage.setItem(SESSION_KEY, JSON.stringify(session));
  }

  function clearSession() {
    localStorage.removeItem(SESSION_KEY);
  }

  function initFirebaseIfAvailable() {
    try {
      if (!window.firebase) return null;
      if (!window.firebaseConfig) return null;
      const cfg = window.firebaseConfig;
      if (!cfg.apiKey || String(cfg.apiKey).includes("YOUR_")) return null;

      // Initialize once
      if (!window.firebase.apps || window.firebase.apps.length === 0) {
        window.firebase.initializeApp(cfg);
      }
      return window.firebase;
    } catch {
      return null;
    }
  }

  async function signInWithGooglePopup() {
    const fb = initFirebaseIfAvailable();
    if (!fb) return null;

    const provider = new fb.auth.GoogleAuthProvider();
    provider.setCustomParameters({ prompt: "select_account" });

    // Best-effort local persistence (ignore if not supported)
    try {
      await fb.auth().setPersistence(fb.auth.Auth.Persistence.LOCAL);
    } catch {}

    const result = await fb.auth().signInWithPopup(provider);
    return result && result.user ? result.user : null;
  }

  function maybeSignOutFromFirebase() {
    const fb = initFirebaseIfAvailable();
    if (!fb) return Promise.resolve();
    return fb.auth().signOut().catch(() => {});
  }

  function toast({ title, message, variant }) {
    const root = document.querySelector(".toasts") || createToastRoot();
    const t = document.createElement("div");
    t.className = "toast" + (variant ? " " + variant : "");
    t.innerHTML = `
      <div class="dot"></div>
      <div>
        <h4>${escapeHtml(title)}</h4>
        <p>${escapeHtml(message || "")}</p>
      </div>
    `;
    root.appendChild(t);
    setTimeout(() => {
      try {
        t.remove();
      } catch {}
    }, 3200);
  }

  function createToastRoot() {
    const root = document.createElement("div");
    root.className = "toasts";
    document.body.appendChild(root);
    return root;
  }

  function escapeHtml(str) {
    return String(str).replace(/[&<>"']/g, (m) => {
      return (
        {
          "&": "&amp;",
          "<": "&lt;",
          ">": "&gt;",
          '"': "&quot;",
          "'": "&#039;",
        }[m] || m
      );
    });
  }

  function parseQuery() {
    const out = {};
    const qs = new URLSearchParams(window.location.search);
    qs.forEach((v, k) => (out[k] = v));
    return out;
  }

  function numericPart(id) {
    const digits = String(id).replace(/\D/g, "");
    return digits ? parseInt(digits, 10) : 0;
  }

  function nextId(items, idKey, prefix) {
    const maxN = items.reduce((acc, it) => Math.max(acc, numericPart(it[idKey]) || 0), 0);
    return prefix + pad4(maxN + 1);
  }

  function readFileAsDataURL(file) {
    return new Promise((resolve, reject) => {
      if (!file) return resolve(null);
      const reader = new FileReader();
      reader.onload = () => resolve(reader.result);
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }

  function requireAuthOrRedirect() {
    const needsAuth = document.body.dataset.auth === "required";
    if (!needsAuth) return;
    const session = getSession();
    if (!session || !session.user_id) {
      toast({
        title: "Login required",
        message: "Please sign in to access this page.",
        variant: "warn",
      });
      window.location.replace("login.html");
    }
  }

  function getOrCreateUserId(identifier, firebase_uid) {
    const users = loadArray(USERS_KEY);
    const normalized = String(identifier).trim();

    // Prefer Firebase UID match when available (prevents duplicates across providers)
    if (firebase_uid) {
      const existingByFirebase = users.find((u) => u.firebase_uid && u.firebase_uid === firebase_uid);
      if (existingByFirebase) return existingByFirebase.user_id;
    }

    const existing = users.find((u) => String(u.identifier).toLowerCase() === normalized.toLowerCase());
    if (existing) return existing.user_id;

    const newUserId = "U" + pad4(users.length ? users.reduce((acc, u) => Math.max(acc, numericPart(u.user_id)), 0) + 1 : 1);
    const user = {
      user_id: newUserId,
      identifier: normalized,
      firebase_uid: firebase_uid || null,
      created_at: nowIso(),
    };
    users.push(user);
    saveArray(USERS_KEY, users);
    return user.user_id;
  }

  function getUserById(user_id) {
    const users = loadArray(USERS_KEY);
    return users.find((u) => u.user_id === user_id) || null;
  }

  function initialsFromId(id) {
    const s = String(id || "");
    if (!s) return "U";
    const parts = s.split(/[^a-zA-Z0-9]+/).filter(Boolean);
    if (parts.length >= 2) return (parts[0][0] + parts[1][0]).toUpperCase();
    return s.slice(0, 2).toUpperCase();
  }

  /* ------------------------------ Event Wiring ------------------------------ */
  async function handleImagePreview(fileInput, previewEl) {
    const file = fileInput.files && fileInput.files[0];
    if (!file) {
      previewEl.innerHTML = `<span style="opacity:.7;font-weight:900;">?</span>`;
      return;
    }
    const data = await readFileAsDataURL(file);
    if (!data) return;
    previewEl.innerHTML = `<img src="${data}" alt="preview" />`;
  }

  async function initIndexPage() {
    const session = getSession();
    const requireLogin = () => {
      if (!session || !session.user_id) {
        toast({ title: "Login first", message: "Sign in to submit lost/found items.", variant: "warn" });
        window.location.replace("login.html");
        return false;
      }
      return true;
    };

    const lostForm = document.getElementById("lost-form");
    const foundForm = document.getElementById("found-form");
    const placeholder = document.getElementById("home-placeholder");
    const lostPanel = document.getElementById("lost-panel");
    const foundPanel = document.getElementById("found-panel");

    function setPanelOpen(which) {
      const showLost = which === "lost";
      const showFound = which === "found";

      if (lostPanel) lostPanel.classList.toggle("is-open", showLost);
      if (foundPanel) foundPanel.classList.toggle("is-open", showFound);
      if (placeholder) placeholder.classList.toggle("is-hidden", showLost || showFound);
    }

    function showFromHash() {
      const hash = String(window.location.hash || "").replace("#", "");
      if (hash === "lost-form") {
        setPanelOpen("lost");
        if (lostPanel) lostPanel.scrollIntoView({ behavior: "smooth", block: "start" });
      } else if (hash === "found-form") {
        setPanelOpen("found");
        if (foundPanel) foundPanel.scrollIntoView({ behavior: "smooth", block: "start" });
      } else {
        setPanelOpen(null);
      }
    }

    // Roll 1-3 requirement: show only the selected form (Lost or Found).
    // Default state shows none unless the user clicks the menu (hash) or navigates with #lost-form/#found-form.
    if (lostPanel && foundPanel && (lostForm || foundForm)) {
      showFromHash();
      window.addEventListener("hashchange", () => showFromHash());

      // Prevent browser from trying to scroll to hidden form nodes
      // before we expand the corresponding collapsible panel.
      const hashAnchors = document.querySelectorAll('a[href$="#lost-form"], a[href$="#found-form"]');
      hashAnchors.forEach((a) => {
        a.addEventListener("click", (e) => {
          const href = a.getAttribute("href") || "";
          const hash = href.split("#")[1];
          if (!hash) return;
          e.preventDefault();
          window.location.hash = hash;
        });
      });
    }

    const lostImgInput = document.getElementById("lost-image");
    const foundImgInput = document.getElementById("found-image");
    const lostPreview = document.getElementById("lost-preview");
    const foundPreview = document.getElementById("found-preview");

    if (lostImgInput && lostPreview) {
      lostImgInput.addEventListener("change", () => handleImagePreview(lostImgInput, lostPreview));
    }
    if (foundImgInput && foundPreview) {
      foundImgInput.addEventListener("change", () => handleImagePreview(foundImgInput, foundPreview));
    }

    if (lostForm) {
      lostForm.addEventListener("submit", async (e) => {
        e.preventDefault();
        if (!requireLogin()) return;

        const user_id = session.user_id;
        const item_name = lostForm.querySelector('[name="item_name"]').value.trim();
        const description = lostForm.querySelector('[name="description"]').value.trim();
        const location = lostForm.querySelector('[name="location"]').value.trim();
        const date = lostForm.querySelector('[name="date"]').value.trim();
        const imageFile = lostForm.querySelector('[name="lost_image"]').files[0] || null;
        const image_data_url = imageFile ? await readFileAsDataURL(imageFile) : null;

        if (!item_name || !description || !location || !date) {
          toast({ title: "Missing fields", message: "Please fill required lost fields.", variant: "danger" });
          return;
        }

        const lostItems = loadArray(LOST_KEY);
        const lost_id = nextId(lostItems, "lost_id", "L");
        lostItems.push({
          lost_id,
          user_id,
          item_name,
          description,
          location,
          date,
          image_data_url,
          created_at: nowIso(),
        });
        saveArray(LOST_KEY, lostItems);

        toast({ title: "Lost item submitted successfully", message: "Saved locally for demo.", variant: "" });
        lostForm.reset();
        if (lostPreview) lostPreview.innerHTML = `<span style="opacity:.7;font-weight:900;">?</span>`;

        // Requirement: after submit, return to dashboard
        window.location.replace("dashboard.html");
      });
    }

    if (foundForm) {
      foundForm.addEventListener("submit", async (e) => {
        e.preventDefault();
        if (!requireLogin()) return;

        const user_id = session.user_id;
        const item_name = foundForm.querySelector('[name="item_name"]').value.trim();
        const description = foundForm.querySelector('[name="description"]').value.trim();
        const location = foundForm.querySelector('[name="location"]').value.trim();
        const date = foundForm.querySelector('[name="date"]').value.trim();
        const imageFile = foundForm.querySelector('[name="found_image"]').files[0] || null;
        const image_data_url = imageFile ? await readFileAsDataURL(imageFile) : null;

        if (!item_name || !description || !location || !date || !imageFile) {
          toast({
            title: "Missing fields",
            message: "Please fill required found fields and select an image.",
            variant: "danger",
          });
          return;
        }

        const foundItems = loadArray(FOUND_KEY);
        const found_id = nextId(foundItems, "found_id", "F");
        foundItems.push({
          found_id,
          user_id,
          item_name,
          description,
          location,
          date,
          image_data_url,
          created_at: nowIso(),
        });
        saveArray(FOUND_KEY, foundItems);

        toast({ title: "Found item submitted successfully", message: "Saved locally for demo.", variant: "" });
        foundForm.reset();
        if (foundPreview) foundPreview.innerHTML = `<span style="opacity:.7;font-weight:900;">?</span>`;

        // Requirement: after submit, return to dashboard
        window.location.replace("dashboard.html");
      });
    }

    // Quick nav to dashboard after submit (optional UX)
    const dashBtn = document.getElementById("go-dashboard");
    if (dashBtn) {
      dashBtn.addEventListener("click", () => {
        if (!requireLogin()) return;
        window.location.href = "dashboard.html";
      });
    }
  }

  function renderEmptyState(el, title, desc) {
    el.innerHTML = `
      <div style="padding:16px;border-radius:16px;border:1px dashed rgba(255,255,255,.18);background:rgba(0,0,0,.12);">
        <div style="font-weight:900;letter-spacing:.2px;">${escapeHtml(title)}</div>
        <div style="margin-top:8px;color:rgba(234,240,255,.65);font-size:13px;line-height:1.5;">${escapeHtml(desc)}</div>
      </div>
    `;
  }

  function initDashboardPage() {
    const session = getSession();
    if (!session || !session.user_id) return;
    const user = getUserById(session.user_id);

    const profileId = document.getElementById("profile-user-id");
    const profileIdentifier = document.getElementById("profile-identifier");
    if (profileId) profileId.textContent = session.user_id;
    if (profileIdentifier) profileIdentifier.textContent = user?.identifier || session.identifier || "";

    const myLostList = document.getElementById("myLostList");
    const myFoundList = document.getElementById("myFoundList");
    const myMatchesList = document.getElementById("myMatchesList");

    const lostItems = loadArray(LOST_KEY).filter((x) => x.user_id === session.user_id);
    const foundItems = loadArray(FOUND_KEY).filter((x) => x.user_id === session.user_id);
    const matches = loadArray(MATCHES_KEY).filter((m) => m.lost_user_id === session.user_id || m.found_user_id === session.user_id);

    if (lostItems.length === 0) {
      renderEmptyState(myLostList, "No lost items yet", "Submit a lost report from the Home page.");
    } else {
      myLostList.innerHTML = "";
      lostItems
        .slice()
        .sort((a, b) => (b.created_at || "").localeCompare(a.created_at || ""))
        .forEach((it) => {
          const wrap = document.createElement("div");
          wrap.className = "item-card";
          wrap.innerHTML = `
            <div class="thumb">
              ${it.image_data_url ? `<img src="${it.image_data_url}" alt="lost image">` : `<div style="width:100%;height:100%;display:flex;align-items:center;justify-content:center;color:rgba(234,240,255,.65);font-weight:900;">L</div>`}
            </div>
            <div class="item-meta">
              <h3>${escapeHtml(it.item_name)}</h3>
              <p>${escapeHtml(it.location)} · ${escapeHtml(it.date)}<br/>${escapeHtml(it.description).slice(0, 110)}${it.description.length > 110 ? "..." : ""}</p>
            </div>
            <div class="item-actions">
              <div class="badge muted">${escapeHtml(it.lost_id)}</div>
            </div>
          `;
          myLostList.appendChild(wrap);
        });
    }

    if (foundItems.length === 0) {
      renderEmptyState(myFoundList, "No found items yet", "Submit a found report from the Home page.");
    } else {
      myFoundList.innerHTML = "";
      foundItems
        .slice()
        .sort((a, b) => (b.created_at || "").localeCompare(a.created_at || ""))
        .forEach((it) => {
          const wrap = document.createElement("div");
          wrap.className = "item-card";
          wrap.innerHTML = `
            <div class="thumb">
              ${it.image_data_url ? `<img src="${it.image_data_url}" alt="found image">` : `<div style="width:100%;height:100%;display:flex;align-items:center;justify-content:center;color:rgba(234,240,255,.65);font-weight:900;">F</div>`}
            </div>
            <div class="item-meta">
              <h3>${escapeHtml(it.item_name)}</h3>
              <p>${escapeHtml(it.location)} · ${escapeHtml(it.date)}<br/>${escapeHtml(it.description).slice(0, 110)}${it.description.length > 110 ? "..." : ""}</p>
            </div>
            <div class="item-actions">
              <div class="badge muted">${escapeHtml(it.found_id)}</div>
            </div>
          `;
          myFoundList.appendChild(wrap);
        });
    }

    if (matches.length === 0) {
      renderEmptyState(myMatchesList, "No AI matches yet", "Once backend AI matching is connected, suggested matches will show here.");
    } else {
      myMatchesList.innerHTML = "";
      const lostById = Object.fromEntries(loadArray(LOST_KEY).map((x) => [x.lost_id, x]));
      const foundById = Object.fromEntries(loadArray(FOUND_KEY).map((x) => [x.found_id, x]));

      matches
        .slice()
        .sort((a, b) => (b.created_at || "").localeCompare(a.created_at || ""))
        .forEach((m) => {
          const lost = lostById[m.lost_id] || {};
          const found = foundById[m.found_id] || {};
          const score = typeof m.score === "number" ? m.score : 0;
          const status = m.status || "pending";
          const statusClass = status === "approved" ? "good" : status === "rejected" ? "warn" : "";

          const wrap = document.createElement("div");
          wrap.className = "item-card";
          wrap.innerHTML = `
            <div class="thumb">
              ${
                lost.image_data_url
                  ? `<img src="${lost.image_data_url}" alt="match preview">`
                  : found.image_data_url
                    ? `<img src="${found.image_data_url}" alt="match preview">`
                    : `<div style="width:100%;height:100%;display:flex;align-items:center;justify-content:center;color:rgba(234,240,255,.65);font-weight:900;">★</div>`
              }
            </div>
            <div class="item-meta">
              <h3>${escapeHtml(lost.item_name || "Lost Item")} ↔ ${escapeHtml(found.item_name || "Found Item")}</h3>
              <p>
                Score: ${(score * 100).toFixed(0)}% · Status: ${escapeHtml(status)}<br/>
                ${escapeHtml(lost.location || "")} → ${escapeHtml(found.location || "")}
              </p>
            </div>
            <div class="item-actions">
              <div class="badge ${statusClass ? statusClass : "muted"}">${escapeHtml(status)}</div>
              <a class="btn" href="match.html?match_id=${encodeURIComponent(m.match_id)}" style="padding:10px 12px; height:40px;">View Match</a>
            </div>
          `;
          myMatchesList.appendChild(wrap);
        });
    }

    const logoutBtn = document.getElementById("logout-btn");
    if (logoutBtn) {
      logoutBtn.addEventListener("click", () => {
        clearSession();
        toast({ title: "Signed out", message: "See you soon.", variant: "" });
        maybeSignOutFromFirebase();
        window.location.replace("login.html");
      });
    }
  }

  function ensureChatForMatch(match, currentUserId) {
    const chats = loadArray(CHATS_KEY);
    const existing = chats.find((c) => c.match_id === match.match_id);
    if (existing) return existing.chat_id;

    // Create a new chat
    const user1_id = match.lost_user_id;
    const user2_id = match.found_user_id;
    // Generate deterministic order
    const [a, b] = user1_id < user2_id ? [user1_id, user2_id] : [user2_id, user1_id];
    const chatId = nextId(chats, "chat_id", "C");
    chats.push({
      chat_id: chatId,
      match_id: match.match_id,
      user1_id: a,
      user2_id: b,
      created_at: nowIso(),
    });
    saveArray(CHATS_KEY, chats);
    return chatId;
  }

  function initMatchPage() {
    const session = getSession();
    if (!session || !session.user_id) return;
    const currentUserId = session.user_id;
    const q = parseQuery();
    const matchId = q.match_id;

    const match = loadArray(MATCHES_KEY).find((m) => m.match_id === matchId);
    const root = document.getElementById("match-root");
    if (!root) return;

    if (!match) {
      root.innerHTML = `
        <div style="padding:16px;">
          <div style="font-weight:950;font-size:18px;">Match not found</div>
          <div style="margin-top:10px;color:rgba(234,240,255,.65);line-height:1.6;font-size:13.5px;">
            AI matching is not connected yet, so there are no saved matches to display.
          </div>
          <div style="margin-top:14px;">
            <a class="btn primary" href="dashboard.html">Go to Dashboard</a>
          </div>
        </div>
      `;
      return;
    }

    const lostItems = loadArray(LOST_KEY);
    const foundItems = loadArray(FOUND_KEY);
    const lost = lostItems.find((x) => x.lost_id === match.lost_id) || null;
    const found = foundItems.find((x) => x.found_id === match.found_id) || null;

    const canChat = match.lost_user_id === currentUserId || match.found_user_id === currentUserId;

    const score = typeof match.score === "number" ? match.score : 0;
    const pct = Math.round(score * 100);
    const scoreBar = document.querySelector(".bar > span");
    if (scoreBar) scoreBar.style.width = Math.max(0, Math.min(100, pct)) + "%";

    const status = match.status || "pending";
    const statusTextEl = document.getElementById("match-status");
    if (statusTextEl) statusTextEl.textContent = status;

    const canChatBtn = document.getElementById("start-chat-btn");
    if (canChatBtn) {
      canChatBtn.disabled = !canChat;
      canChatBtn.style.opacity = canChat ? "1" : ".55";
      canChatBtn.style.cursor = canChat ? "pointer" : "not-allowed";
    }

    const lostTitle = document.getElementById("lost-item-name");
    const foundTitle = document.getElementById("found-item-name");
    if (lostTitle) lostTitle.textContent = lost?.item_name || "Lost item";
    if (foundTitle) foundTitle.textContent = found?.item_name || "Found item";

    const setText = (id, val) => {
      const el = document.getElementById(id);
      if (el) el.textContent = val || "";
    };

    setText("lost-location", lost?.location);
    setText("lost-date", lost?.date);
    setText("lost-description", lost?.description);

    setText("found-location", found?.location);
    setText("found-date", found?.date);
    setText("found-description", found?.description);

    const lostImg = document.getElementById("lost-image");
    if (lostImg) {
      if (lost?.image_data_url) lostImg.src = lost.image_data_url;
      else lostImg.parentElement.innerHTML = `<div style="opacity:.7;font-weight:950;">No Image</div>`;
    }

    const foundImg = document.getElementById("found-image");
    if (foundImg) {
      if (found?.image_data_url) foundImg.src = found.image_data_url;
      else foundImg.parentElement.innerHTML = `<div style="opacity:.7;font-weight:950;">No Image</div>`;
    }

    const scoreText = document.getElementById("match-score");
    if (scoreText) scoreText.textContent = `${pct}%`;

    if (canChatBtn) {
      canChatBtn.addEventListener("click", () => {
        if (!canChat) return;
        const chatId = ensureChatForMatch(match, currentUserId);
        window.location.href = `chat.html?chat_id=${encodeURIComponent(chatId)}`;
      });
    }
  }

  function initChatPage() {
    const session = getSession();
    if (!session || !session.user_id) return;
    const currentUserId = session.user_id;
    const q = parseQuery();
    const chatId = q.chat_id;

    const chat = loadArray(CHATS_KEY).find((c) => c.chat_id === chatId);
    const root = document.getElementById("chat-root");
    if (!root) return;

    if (!chat) {
      root.innerHTML = `
        <div style="padding:16px;">
          <div style="font-weight:950;font-size:18px;">Chat not found</div>
          <div style="margin-top:10px;color:rgba(234,240,255,.65);line-height:1.6;font-size:13.5px;">
            Start a chat from a valid match first.
          </div>
          <div style="margin-top:14px;">
            <a class="btn primary" href="dashboard.html">Go to Dashboard</a>
          </div>
        </div>
      `;
      return;
    }

    if (chat.user1_id !== currentUserId && chat.user2_id !== currentUserId) {
      toast({ title: "Not authorized", message: "You are not part of this chat.", variant: "danger" });
      window.location.replace("dashboard.html");
      return;
    }

    const match = loadArray(MATCHES_KEY).find((m) => m.match_id === chat.match_id);
    const lostItems = loadArray(LOST_KEY);
    const foundItems = loadArray(FOUND_KEY);
    const lost = match ? lostItems.find((x) => x.lost_id === match.lost_id) : null;
    const found = match ? foundItems.find((x) => x.found_id === match.found_id) : null;

    const otherUserId = chat.user1_id === currentUserId ? chat.user2_id : chat.user1_id;
    const otherUser = getUserById(otherUserId);

    const msgWrap = document.getElementById("messages");
    const composer = document.getElementById("composer-text");
    const proofInput = document.getElementById("proof-image");
    const proofPreview = document.getElementById("proof-preview");
    const sendBtn = document.getElementById("send-btn");

    const chatTitle = document.getElementById("chat-title");
    if (chatTitle) chatTitle.textContent = `Chat • ${lost?.item_name || "Lost"} ↔ ${found?.item_name || "Found"}`;

    const chatSubtitle = document.getElementById("chat-subtitle");
    if (chatSubtitle) {
      const theirId = otherUser?.identifier || otherUserId;
      chatSubtitle.textContent = `You are chatting with ${theirId}.`;
    }

    if (proofInput && proofPreview) {
      proofInput.addEventListener("change", async () => {
        const file = proofInput.files && proofInput.files[0];
        if (!file) {
          proofPreview.innerHTML = `<div style="opacity:.7;font-weight:950;">No Proof</div>`;
          return;
        }
        const data = await readFileAsDataURL(file);
        proofPreview.innerHTML = `<img src="${data}" alt="proof preview"/>`;
      });
    }

    function scrollToBottom() {
      if (!msgWrap) return;
      msgWrap.scrollTop = msgWrap.scrollHeight;
    }

    function renderMessages() {
      if (!msgWrap) return;
      const messages = loadArray(MESSAGES_KEY).filter((m) => m.chat_id === chatId).sort((a, b) => (a.created_at || "").localeCompare(b.created_at || ""));

      const currentSeenUpdate = messages.map((m) => {
        const seen_by = Array.isArray(m.seen_by) ? m.seen_by : [];
        if (seen_by.includes(currentUserId)) return null;
        seen_by.push(currentUserId);
        m.seen_by = seen_by;
        return m;
      }).filter(Boolean);

      if (currentSeenUpdate.length > 0) {
        // Persist seen status
        const all = loadArray(MESSAGES_KEY);
        const byId = new Map(currentSeenUpdate.map((m) => [m.message_id, m]));
        const updated = all.map((m) => (byId.has(m.message_id) ? byId.get(m.message_id) : m));
        saveArray(MESSAGES_KEY, updated);
      }

      msgWrap.innerHTML = "";
      if (messages.length === 0) {
        msgWrap.innerHTML = `
          <div style="opacity:.7;text-align:center;padding:22px 8px;">
            No messages yet. Send the first text to start verification.
          </div>
        `;
        return;
      }

      messages.forEach((m) => {
        const me = m.sender_id === currentUserId;
        const avatar = initialsFromId(me ? currentUserId : otherUserId);
        const msg = document.createElement("div");
        msg.className = "msg" + (me ? " me" : "");

        const ts = (m.created_at || "").replace("T", " ").slice(0, 16);
        const proofHtml = m.image_data_url
          ? `<div class="proof"><img src="${m.image_data_url}" alt="proof"/></div>`
          : "";

        msg.innerHTML = `
          <div class="avatar">${escapeHtml(avatar)}</div>
          <div class="bubble">
            <p>${escapeHtml(m.text || "")}</p>
            ${proofHtml}
            <div class="ts">${escapeHtml(ts || "")}</div>
          </div>
        `;
        msgWrap.appendChild(msg);
      });
      scrollToBottom();
    }

    let lastCount = -1;
    function poll() {
      if (!msgWrap) return;
      const messages = loadArray(MESSAGES_KEY).filter((m) => m.chat_id === chatId);
      if (messages.length !== lastCount) {
        lastCount = messages.length;
        renderMessages();
      }
    }

    renderMessages();
    setInterval(poll, 1500);

    if (sendBtn && composer) {
      sendBtn.addEventListener("click", async () => {
        const text = composer.value.trim();
        const file = proofInput && proofInput.files ? proofInput.files[0] : null;
        const image_data_url = file ? await readFileAsDataURL(file) : null;

        if (!text && !image_data_url) {
          toast({ title: "Empty message", message: "Write a message or attach a proof image.", variant: "warn" });
          return;
        }

        const messages = loadArray(MESSAGES_KEY);
        const message_id = nextId(messages, "message_id", "MSG");

        messages.push({
          message_id,
          chat_id: chatId,
          sender_id: currentUserId,
          text,
          image_data_url,
          created_at: nowIso(),
          seen_by: [currentUserId],
        });
        saveArray(MESSAGES_KEY, messages);

        composer.value = "";
        if (proofInput) proofInput.value = "";
        if (proofPreview) proofPreview.innerHTML = `<div style="opacity:.7;font-weight:950;">No Proof</div>`;
        renderMessages();
      });
    }
  }

  function initCommon() {
    const logoutBtn = document.getElementById("logout-btn");
    if (logoutBtn) {
      logoutBtn.addEventListener("click", () => {
        clearSession();
        toast({ title: "Signed out", message: "See you soon.", variant: "" });
        window.location.replace("login.html");
      });
    }

    const mobileLogoutBtn = document.getElementById("mobile-logout-btn");
    if (mobileLogoutBtn) {
      mobileLogoutBtn.addEventListener("click", () => {
        clearSession();
        toast({ title: "Signed out", message: "See you soon.", variant: "" });
        maybeSignOutFromFirebase();
        window.location.replace("login.html");
      });
    }

    const hamburger = document.getElementById("nav-hamburger");
    const sidebar = document.getElementById("mobile-sidebar");
    const overlay = document.getElementById("mobile-overlay");
    const closeBtn = document.getElementById("mobile-sidebar-close");

    if (hamburger && sidebar && overlay && closeBtn) {
      const setOpen = (isOpen) => {
        if (isOpen) {
          sidebar.classList.add("open");
          overlay.classList.add("open");
          sidebar.setAttribute("aria-hidden", "false");
          overlay.setAttribute("aria-hidden", "false");
          document.body.style.overflow = "hidden";
        } else {
          sidebar.classList.remove("open");
          overlay.classList.remove("open");
          sidebar.setAttribute("aria-hidden", "true");
          overlay.setAttribute("aria-hidden", "true");
          document.body.style.overflow = "";
        }
      };

      hamburger.addEventListener("click", () => setOpen(true));
      closeBtn.addEventListener("click", () => setOpen(false));
      overlay.addEventListener("click", () => setOpen(false));

      // Close on any sidebar link click (better UX on mobile)
      const sidebarLinks = sidebar.querySelectorAll("a, button");
      sidebarLinks.forEach((el) => {
        el.addEventListener("click", () => setOpen(false));
      });

      window.addEventListener("keydown", (e) => {
        if (e.key === "Escape") setOpen(false);
      });
    }

    // Prevent full page reload on hash changes for smooth UX (optional)
    window.addEventListener("click", (e) => {
      const a = e.target.closest("a");
      if (!a) return;
      const href = a.getAttribute("href") || "";
      if (href.startsWith("#")) {
        // Let browser handle anchors
        return;
      }
    });
  }

  function boot() {
    requireAuthOrRedirect();
    initCommon();

    if (document.getElementById("lost-form") || document.getElementById("found-form")) {
      initIndexPage().catch(() => {});
    }

    if (document.getElementById("profile-user-id") || document.getElementById("myLostList")) {
      initDashboardPage();
    }

    if (document.getElementById("match-root")) {
      initMatchPage();
    }

    if (document.getElementById("chat-root")) {
      initChatPage();
    }

    const loginForm = document.getElementById("login-form");
    if (loginForm) {
      loginForm.addEventListener("submit", async (e) => {
        e.preventDefault();
        const identifier = loginForm.querySelector('[name="email_or_student_id"]').value.trim();
        if (!identifier) {
          toast({ title: "Invalid input", message: "Enter your email or student ID.", variant: "danger" });
          return;
        }

        let fbUser = null;
        try {
          fbUser = await signInWithGooglePopup();
        } catch {
          toast({ title: "Sign-in cancelled", message: "Firebase Google sign-in did not complete.", variant: "warn" });
          return;
        }

        // If Firebase is configured, use Firebase UID to create/reuse the local profile.
        if (fbUser && fbUser.uid) {
          const firebase_uid = fbUser.uid;
          const user_id = getOrCreateUserId(identifier, firebase_uid);
          setSession({ user_id, identifier, firebase_uid });
          toast({ title: "Login successful", message: "Welcome back.", variant: "" });
          window.location.replace("dashboard.html");
          return;
        }

        // Fallback: if Firebase is not configured, keep the original Roll 1–3 local login.
        const user_id = getOrCreateUserId(identifier, null);
        setSession({ user_id, identifier, firebase_uid: null });
        toast({
          title: "Login successful (local demo)",
          message: "Firebase config is missing/unset. Replace `frontend/js/firebaseConfig.js` to enable real login.",
          variant: "warn",
        });
        window.location.replace("dashboard.html");
      });
    }
  }
